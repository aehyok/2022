mongodb 关联查询(通过主文档字段查询) ​

const data = await Chat.aggregate([
   {
      $match: {
         userId: new mongoose.Types.ObjectId(userId)
      }
   },
   {
      $lookup: {
         from: 'models',
         localField: 'modelId',
         foreignField: '_id',
         as: 'modelInfo'
      }
   },
   {
      $project: {
         _id: 1, // 设置为 0 表示不显示
         title: 1, // 设置为 1 表示显示
         modelId: 1, // 设置为 1 表示显示
         updateTime: 1, // 设置为 1 表示显示
         latestChat: 1,
         'modelInfo.name': 1,
         'modelInfo.remark': 1
      }
   },
   { $sort: { updateTime: -1 } },
   { $limit: 5 }
   ]);
